Less code for more plots

line_plot = function(feature) {
  p = ggplot(data = data_reshape[data_reshape$feature == feature, ], aes(x = frame_index/1000, y = value)) +
    geom_line(size = 0.1) +
    xlab('') +
    ggtitle(feature_names[feature]) +
    theme(plot.title = element_text(size = 5, face = "bold"))
  return(p)
}

density_plot = function(feature) {
  p = ggplot(data = data_reshape[data_reshape$feature == feature, ], aes(x = value)) +
    geom_density(fill = 'blue') + coord_flip() +
    xlab('') + ylab('') +
    ggtitle('') +
    theme(legend.position = 'none', plot.title = element_text(size = 5, face = "bold")) 
  return(p)
}

line_plots = map(names(feature_names), line_plot)
density_plots = map(names(feature_names), density_plot)

all_plots = vector(length = length(line_plots) + length(density_plots)  )
all_plots[seq(1, 29*2, 2)] = line_plots
all_plots[seq(2, 29*2, 2)] = density_plots

all_plots$ncol = 6
all_plots$widths = c(5,2,5,2,5,2)
do.call(grid.arrange,  all_plots)
